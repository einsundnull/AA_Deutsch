> panel-renderer.js <- Kann ich diese Datei als die Basisklasse für ein Panel verstehen?                                
● Ja, aber mit einer wichtigen Klarstellung:

  panel-renderer.js ist NICHT eine echte Basisklasse, sondern ein "Renderer-Service":                                   
  ┌─────────────────────────────────────┐                                                                                 │   Panel-Daten (Group-Objekt)        │  ← Reine Daten-Struktur
  │   {id, status, dayIndex, ...}       │
  └─────────────────────────────────────┘
              ↓
  ┌─────────────────────────────────────┐
  │   PanelRenderer                     │  ← Renderer-Service
  │   (panel-renderer.js)               │     (erstellt HTML-Elemente)
  │                                     │
  │   createPanelElement()              │
  │   applyPanelStyles()                │
  │   attachInteractivePanelListeners() │
  └─────────────────────────────────────┘
              ↓
  ┌─────────────────────────────────────┐
  │   Rendered Panel (DOM-Element)      │  ← HTML-Element im Browser
  │   <div class="time-slot-panel">     │
  └─────────────────────────────────────┘

  Das ist kein klassisches OOP-Vererbungs-Pattern:

  // ❌ NICHT so:
  class Panel { /* Basisklasse */ }
  class BookedPanel extends Panel { }
  class TimeoutPanel extends Panel { }

  // ✅ Stattdessen:
  class PanelRenderer {
      createPanelElement(group) {  // Ein Status-Parameter, nicht mehrere Subklassen
          switch(group.status) {
              case 'booked': /* ... */
              case 'timeout': /* ... */
          }
      }
  }

     Plan: Panel-Renderer Refaktor (Kompletter Replace)
                                                                                                                             Context
                                                                                                                             panel-renderer.js ist ein monolithischer Renderer-Service (~990 Zeilen), der Status-spezifisches Verhalten über
     If/Else und Maps steuert. Ziel ist echter OOP-Polymorphismus: jeder Panel-Status bekommt eine eigene Klasse, die
     ihre Spezifika selbst definiert.

     Warum: Status-Logik ist verstreut (Text, Listener, Delete-Config, Badges), schwer zu erweitern. Mit Subklassen:
     neuer Status = neue Datei, kein Anfassen der Basis.

     panel-available.js     ← AvailablePanel extends PanelBase (blocked, available)
     panel-renderer.js      ← KOMPLETT NEU: Overlay-Koordinator + Factory-Delegation

     1. panel-base.js — Abstrakte Basisklasse

     Kapselt: ein einzelnes Panel-DOM-Element bauen, stylen, Listener anhängen.

     export class PanelBase {
         constructor(group, renderer) {
             this.group = group;
             this.renderer = renderer;       // PanelRenderer-Instanz
             this.calendar = renderer.calendar;
             this.grouper = renderer.grouper;
             this.panelState = renderer.panelState;
         }

         // GEMEINSAME FACTORY-METHODE (aufgerufen von PanelRenderer)
         createElement(calendarBody) {
             const firstCell = this.renderer.findTableCell(calendarBody, this.group.dayIndex, this.group.startSlot);
             const lastCell  = this.renderer.findTableCell(calendarBody, this.group.dayIndex, this.group.endSlot);
             if (!firstCell || !lastCell) return null;

             const panel = document.createElement('div');
             this._applyDataset(panel);
             this._applyPosition(panel, firstCell, lastCell);
             this._buildChildren(panel);           // gemeinsam: delete/topHandle/resizeHandle/spinner/text
             this._buildStatusChildren(panel);     // hook für Subklassen (z.B. ConfirmationBadge)
             this.renderer.applyPanelStyles(panel, this.renderer.calculatePanelDimensions(firstCell, lastCell));
             this.renderer.applyCellStylingToPanel(panel, this.group.status, this.group.displayText);
             this.attachListeners(panel, calendarBody); // abstrakt — Subklasse implementiert
             return panel;
         }

         // GEMEINSAM
         _applyDataset(panel) { /* dataset.panelId, dayIndex, status, date, teacherUid … */ }
         _applyPosition(panel, firstCell, lastCell) { /* cssText */ }
         _buildChildren(panel) { /* deleteButton, topHandle, resizeHandle, spinner, textSpan */ }

         // HOOKS FÜR SUBKLASSEN
         _buildStatusChildren(panel) {}           // Standard: kein extra Child
         attachListeners(panel, calendarBody) {}  // Standard: keine Listener

         // GEMEINSAME HILFSMETHODEN (Subklassen erben)
         buildText() { return this.group.displayText; }
         getDeleteConfig() {
             return { chatEnabled: false, studentUIDs: [], operation: 'deletion' };
         }
         // createDeleteButton(), createTopHandle(), createResizeHandle(), createLoadingSpinner()
         // createPanelText(), createConfirmationBadge()  ← unverändert aus panel-renderer.js
     }

     import { BookedPanel }    from './panel-booked.js';
     import { PersonalPanel }  from './panel-personal.js';
     import { TimeoutPanel }   from './panel-timeout.js';
     import { AvailablePanel } from './panel-available.js';

     const PANEL_CLASS_MAP = {
         booked:    BookedPanel,
         personal:  PersonalPanel,
         timeout:   TimeoutPanel,
         available: AvailablePanel,
         blocked:   AvailablePanel,
     };

     export class PanelRenderer {
         constructor(grouper) { /* unverändert */ }

         createPanelElement(group, calendarBody) {
             // PersonalPanel-Guard
             if (group.status === 'personal' && this.calendar.state.preventPersonalPanel) return null;

             const PanelClass = PANEL_CLASS_MAP[group.status] ?? AvailablePanel;
             const panelObj   = new PanelClass(group, this);
             return panelObj.createElement(calendarBody);
         }

         // ALLE FOLGENDEN METHODEN BLEIBEN UNVERÄNDERT (externe API):
         // createOverlayContainer(), renderAllPanels(), removePanelFromUI()
         // applyCellStylingToPanel(), findTableCell(), calculatePanelDimensions()

         // createOverlayContainer(), removeOverlayContainer(), attachOverlayHoverListeners()

         // LÖSCH-LOGIK BLEIBT IN PanelRenderer (Aufrufer: deleteButton-Listener in PanelBase)
         // deletePanel(), deletePanelUnified(), removePanelFromUI()
         // showDeleteConfirmationModal(), createDeleteModal(), closeDeleteModal()
         // showPastBookingWarning(), isPanelInPast()
         // getStatusDisplayText(), createSlotsFromGroup()

         // GEMEINSAME ELEMENT-BUILDER (aufgerufen von PanelBase)
